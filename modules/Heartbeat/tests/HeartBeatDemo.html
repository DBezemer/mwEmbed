Heartbeat Plugin Configuration - Adding Adobe Heartbeat analytics plugin


Adding Adobe Heartbeat analytics plugin

You can configure Adobe Heartbeat plugin inside the heartbeat plugin flashvar.


Sample code for Heartbeat configuration: 
        kWidget.featureConfig({
                'targetId': 'kdp',
                'wid': '_243342',
                'uiconf_id' : '21099702',
                'entry_id' : '1_sf5ovm7u',
                'readyCallback': onReadyCallback,
                'flashvars': {
                    'heartbeat': {
                        plugin: true,

                        visitorId: '9A32525A5492AE7C0A4C98C6@AdobeOrg',
                        visitorTrackingServer: 'ovppartners.d1.sc.omtrdc.net',

                        appMeasurmentAccount: 'tester',
                        appMeasurmentTrackingServer: 'ovppartners.d1.sc.omtrdc.net'
                        appMeasurmentPageName: 'Kaltura Test Page',
                        appMeasurmentCharSet: 'UTF-8',
                        appMeasurmentVisitorID: 'test-vid',

                        heartbeatTrackingServer: 'ovppartners.d1.sc.omtrdc.net',
                        heartbeatPublisher: 'Kaltura test-publisher'
                    }
                }
            });
